50/*
51 * Device stream abstraction for either an internal or external stream.
52 * Used for both Unicam and the ISP.
53 */
54class Stream : public libcamera::Stream
55{
56public:
57 enum class StreamFlag {
58 None = 0,
59 /*
60 * Indicates that this stream only imports buffers, e.g. the ISP
61 * input stream.
62 */
63 ImportOnly = (1 << 0),
64 /*
65 * Indicates that this stream is active externally, i.e. the
66 * buffers might be provided by (and returned to) the application.
67 */
68 External = (1 << 1),
69 /*
70 * Indicates that the stream buffers need to be mmaped and returned
71 * to the pipeline handler when requested.
72 */
73 RequiresMmap = (1 << 2),
74 /*
75 * Indicates a stream that needs buffers recycled every frame internally
76 * in the pipeline handler, e.g. stitch, TDN, config. All buffer
77 * management will be handled by the pipeline handler.
78 */
79 Recurrent = (1 << 3),
80 /*
81 * Indicates that the output stream needs a software format conversion
82 * to be applied after ISP processing.
83 */
84 Needs32bitConv = (1 << 4),
85 };
86
87 using StreamFlags = Flags<StreamFlag>;

154 * List of frame buffers that we can use if none have been provided by
155 * the application for external streams. This is populated by the
156 * buffers exported internally.
157 */
158 std::queue<FrameBuffer *> availableBuffers_;
159
160 /*
161 * List of frame buffers that are to be queued into the device from a Request.
162 * A nullptr indicates any internal buffer can be used (from availableBuffers_),
163 * whereas a valid pointer indicates an external buffer to be queued.
164 *
165 * Ordering buffers to be queued is important here as it must match the
166 * requests coming from the application.
167 */
168 std::queue<FrameBuffer *> requestBuffers_;
169
170 /*

172 * as the stream needs to maintain ownership of these buffers.
173 */
174 std::vector<std::unique_ptr<FrameBuffer>> internalBuffers_;

16/*
17 * Camera metadata parser class. Usage as shown below.
18 *
19 * Setup:
20 *
21 * Usually the metadata parser will be made as part of the CamHelper class so
22 * application code doesn't have to worry which kind to instantiate. But for
23 * the sake of example let's suppose we're parsing imx219 metadata.
24 *
25 * MdParser *parser = new MdParserSmia({ expHiReg, expLoReg, gainReg });
26 * parser->SetBitsPerPixel(bpp);
27 * parser->SetLineLengthBytes(pitch);
28 * parser->SetNumLines(2);
29 *
30 * Note 1: if you don't know how many lines there are, the size of the input
31 * buffer is used as a limit instead.
32 *
33 * Note 2: if you don't know the line length, you can leave the line length unset
34 * (or set to zero) and the parser will hunt for the line start instead.
35 *
36 * Then on every frame:
37 *
38 * RegisterMap registers;
39 * if (parser->Parse(buffer, registers) != MdParser::OK)
40 * much badness;
41 * Metadata metadata;
42 * CamHelper::PopulateMetadata(registers, metadata);
43 *
44 * (Note that the CamHelper class converts to/from exposure lines and time,
45 * and gain_code / actual gain.)
46 *
47 * If you suspect your embedded data may have changed its layout, change any line
48 * lengths, number of lines, bits per pixel etc. that are different, and
49 * then:
50 *
51 * parser->Reset();
52 *
53 * before calling Parse again.
54 */
55

133 /*
134 * Note that error codes > 0 are regarded as non-fatal; codes < 0
135 * indicate a bad data buffer. Status codes are:
136 * ParseOk - found all registers, much happiness
137 * MissingRegs - some registers found; should this be a hard error?
138 * The remaining codes are all hard errors.
139 */
140 enum ParseStatus {
141 ParseOk = 0,
142 MissingRegs = 1,
143 NoLineStart = -1,
144 IllegalTag = -2,
145 BadDummy = -3,
146 BadLineEnd = -4,
147 BadPadding = -5
148 };

152 OffsetMap offsets_;
153};
